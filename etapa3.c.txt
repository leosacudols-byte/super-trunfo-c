#include <stdio.h>

int main() {
    int ataque1 = 80, defesa1 = 70, velocidade1 = 90;
    int ataque2 = 75, defesa2 = 85, velocidade2 = 85;

    int escolha1, escolha2;
    int total1 = 0, total2 = 0;

    printf("=== SUPER TRUNFO ===\n");
    printf("Escolha o primeiro atributo:\n");
    printf("1 - Ataque\n2 - Defesa\n3 - Velocidade\n>> ");
    scanf("%d", &escolha1);

    printf("Escolha o segundo atributo:\n");
    printf("1 - Ataque\n2 - Defesa\n3 - Velocidade\n>> ");
    scanf("%d", &escolha2);

    // Função auxiliar com switch para atribuir valores
    switch(escolha1) {
        case 1: total1 += ataque1; total2 += ataque2; break;
        case 2: total1 += defesa1; total2 += defesa2; break;
        case 3: total1 += velocidade1; total2 += velocidade2; break;
        default: printf("Atributo inválido!\n"); return 0;
    }
    switch(escolha2) {
        case 1: total1 += ataque1; total2 += ataque2; break;
        case 2: total1 += defesa1; total2 += defesa2; break;
        case 3: total1 += velocidade1; total2 += velocidade2; break;
        default: printf("Atributo inválido!\n"); return 0;
    }

    printf("\nComparando os dois atributos escolhidos...\n");
    (total1 > total2) ? printf("Carta 1 venceu com %d pontos!\n", total1) :
    (total2 > total1) ? printf("Carta 2 venceu com %d pontos!\n", total2) :
                        printf("Empate com %d pontos!\n", total1);

    return 0;
}
